# Recommend

Get work or manifestation recommendations based on a pid.

  query={`query ($pid: String!, $limit: Int!) {

          main
        }
        creators {
          display
        }
      }
    }
  }
}`}
  variables={{
    pid: "870970-basis:55139784",
    limit: 10,
