## Submit remit Order

Create an interlibrary loan based on an enduser loan.

The only difference between an order and a remit order, is the originalOrderId field.

The remit order must have all the data from the original order and its orderId in the originalOrderId field.

<InlineGraphiQL
  query={`mutation example_submitremitorder($input: NetpunktSubmitOrderInput!) {
    netpunkt {
      submitOrder(input: $input, dryRun: true) {
        status
        message
        orderId
      }
    }

          }`}

variables={{
    input: {
      pids: ["870970-basis:25574486"],
      pickUpAgencyId: "790900",
      originalOrderId: "123456",
      userParameters: {
        userId: "0102033690",
        userAddress: "test",
        userName: "Flemming test",
        userMail: "flemminghoimark@gmail.com",
      },
    },
  }}
/>

## Close Order

<InlineGraphiQL
  query={`mutation example_closeorder($input: NetpunktCloseOrderInput!) {

        orderId
      }
    }

          }`}

variables={{
